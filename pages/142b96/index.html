mysql笔记整理

# 前言

数据库有很多种管理软件，mysql应该算是最次的那种，因为这种数据库安全性最低，比较好的数据库有：Mysqli、PDO这种，因为它们都有预编译的限制，导致注入更加困难。

# 背景

为什么要学习mysql了？因为它安全性低，现在ctf考试也基本就考它，就跟web考php道理差不多……

不过学长说，之后的ctf比赛会越来越难，越来越偏，之后其他数据库的内容也肯定会逐渐进入赛场。不过sql语句在大体上都是差不多的，只有一些细节上的区别，所以还是先学好mysql。

# 环境配置

自己下载安装也可以，但是集成化的服务器搭建软件也一般有集成mysql，我用的phpstudy（好嘛，两个最不安全的东西集成到一起了），里面就内置了mysql，还有其他集成软件也有，可以自行下载安装。

# 数据库结构

数据库的结构很简单，基本上很快就可以掌握，但不能弄错。

<(库名1)database> ——<(表格名1)table>
                 |——<(表格名2)table>
                 |——<(表格名3)table>
<(库名2)database> ——<(表格名1)table>
                 |——<(表格名2)table>
   …………
   
#就这么简单，每个表格就是一个二维表，分为行和列(column)


#除此以外，mysql一般都会自带4个库：
==information_schema==
一个信息数据库，它保存着关于MySQL服务器所维护的所有其他数据库的信息

==Mysql==
mysql的核心数据库，类似于sql server中的master表，主要负责存储数据库的用户、权限设置、关键字等mysql自己需要使用的控制和管理信息

==performance_schema==
主要用于收集数据库服务器性能参数。并且库里表的存储引擎均为PERFORMANCE_SCHEMA，而用户是不能创建存储引擎为PERFORMANCE_SCHEMA的表。MySQL5.7默认是开启的

==Sys==
sys库所有的数据源来自performance_schema。目标是把performance_schema的把复杂度降低，让DBA能更好的阅读这个库里的内容

#后两个对于咱们来说基本没什么用，前两个是渗透必须了解的数据库。
#information_schema保存着其他数据库的信息，可以说爆破出它的信息我们就能去爆破其他库的信息，结构如下：

# sql基本语句

需要注意的是，mysql的语句需要加分号;，初学者常忘记。 同时，sql语句不区分大小写。

反引号可以在语句操作中包裹库名或者表名，只是为了区分它们和数据库语句，字母名可以不用，但是如果是数字名的话就必须加反引号包裹，反引号中不能用点.连接，应该如：information_schema.tables

这样，同时里面不能有函数，函数可以用括号包裹

ps：学习语句不是主要内容，更重要的是学到sql关键词的使用方法，语句的使用也不过是==关键词的灵活调用==，那才是关键。 sql关键词整理跳转

#显示存在的数据库
Show databases; #注意加了s

#使用指定数据库
Use test;

#显示当前数据库存在的表
Show tables; #注意加了s，并且使用此操作必须先use进入某个数据库

#查询指定表的所有数据（users即为表名）
Select * from users;

#查询指定表的某些行（users为表名，id为列名，1为列值）
Select * from users where id=1;

#只查询指定表的某些列（users为表名，username和password为列名）
Select username,password from users;

#查询指定表每一列的基本信息或详细信息
show [full] columns from <tablename>

#创建数据库
Create database <test>;

#创建表（需要指定列数和列的属性）
CREATE TABLE `users`(
   `id` INT AUTO_INCREMENT PRIMARY KEY,
   `username` VARCHAR(100) NOT NULL,
   `password` VARCHAR(100) NOT NULL
);
#反引号`的使用是为了区分表名和关键字，其实用不用都可。

#此语句中用了两个参数auto_increment是自动递增，此时赋值时给它赋NULL它便会自行递增，如果是第一个则会赋值为1，一般在id这个参数上使用；primary key，意思就是可以设定这个属性为可以识别这个对象，即表中的每一个对象此属性都必须不同，都是独一无二的。

#Mysql数据类型：https://www.runoob.com/mysql/mysql-data-types.html


#插入一行数据
INSERT INTO table_name(id,username,password)VALUES( NULL,'LionTree','hahahaha');
insert into table_name(id,username,password)values( null,'liontree','hahahaha');

#更新数据(username为列名)
UPDATE users SET username='admin' WHERE id=13;
update users set username='admin' where id=13;

#删除数据(users为表名)
DELETE FROM users where id=1;
delete from users where id=1;

#添加、删除、修改表的字段（即表中的一列数据）
#删除一列属性
alter table <table_name> drop <列名，即field>;
#添加一列int属性数据栏，并命名为<自定义列名>
alter table <table_name> add <自定义列名> int;
#如果你需要指定新增字段的位置，可以使用关键字first(设定位第一列),after字段名（设定位于某个字段之后）。在执行成功后，使用show columns查看表结构的变化:
alter table <table_name> add <自定义列名> int first;
alter table <table_name> add <自定义列名> int after <列名>;
#修改字段类型及名称，两种方式，modify和change，语法区别较大
#例如，把字段 c 的类型从 CHAR(1) 改为 CHAR(10),c即为列名
alter table <table_name> modify c CHAR(10);
#在change关键字之后，紧跟着的是你要修改的字段名，然后指定新字段名及类型。如下将i字段切换为j字段并修改类型为bigint
alter table <table_name> CHANGE i j BIGINT;

#删除数据表
drop table <table_name>;

#删除数据库
drop database <database_name>;

select * from <表名1> union select * from <表名2>;
#联合查找，即讲两张表合并在一起然后显示
#要求：列数相同，否则会报错
#union select不仅可以选择其他表中的数据与该表进行结合，而且还可以“自己添加对象”，比如有3列，就可以union select 1,2,3; 此时显示时，就会将表中原数据和属性分别为123的对象也显示出来。更重要的是，其中的属性换为sql语句，即注入漏洞，下文有讲解。

#以那一列的数据为判断条件，递增序列排序显示
select * from <表名> order by <列名>;

#递减序列排序
select * from <表名> order by <列名> desc;

select * from student order by 3;
#3代表第3列，即以第3列为判断条件，递增排序显示
#如果网页有回显，可以用这个语句检测表对的列数

#设置变量a=1
set @a=1;

#预处理语句，创建语句，执行，释放：
#创建语句，xiaohua为语句名，可随便改，'SELECT ?+?'则为mysql语句，用问号?来代替变量的地方，这个语句不仅仅可以是字符串形式，还可以是16进制形式，可以利用这条属性绕过前端的滤过，因为prepare…from…是预处理语句，会进行编码转换。
prepare xiaohua from 'select ?+?'; 
#execute即为执行，同时用using+变量,变量来代替mysql语句中的问号?，前面已设定set @a=1,@b=10
execute xiaohua using @a,@b;
#释放语句xiaohua
deallocate prepare xiaohua;
#set可以用于设置变量，变量不仅能用于设置数值，还可以直接构建mysql语句，如下（上下相同）：
';SeT@a=0x73656c656374202a2066726f6d20603139313938313039333131313435313460;prepare execsql from @a;execute execsql;#
';sEt @a=concat("sel","ect flag from `1919810931114514`");Prepare hello from @a;execute hello;#

# sql关键词整理

show <databases/tables/columns/full columns>;
#显示全部数据库名称或者某库中全部表格名称，还能显示表的信息
use <库名>;
#使用某数据库
select <列名/*>
#查询某些行数据，星号*是通配符，代表全部
#select后面不能用点.连接
from <表名/库名.表名>
#限定来自那个表，使用库名.表名时需要当前使用的数据库不是那个库，也就是在库外时可以用.连接寻找那个库中的表
where <列名>=<值>;
#限定属性值
create <database/table> <自定义库名/表名>;
#例子：
create table handler_table(  
    c1 int,   
    c2 varchar(10),   
    c3 int(10) 
);  
#创建表的操作还需要在后面加括号()，并在括号中填入定义的列名和列的类型还有属性
insert into <表名>(id,username,password)values( null,'liontree','hahahaha');
#其中的id，username，password都是列名，插入是需要提供列名的，至于id赋值为null是因为id这一列用了auto_increment参数，上文有讲。
delete <form|whele等限定语句>;
#用于删除一行或多行数据
alter table <table_name>
#操作表中的列数据时需要用这个指定表，然后分别用drop和add删除和添加列，用modify或change修改
drop <database/table/行名> <库名/表名>;
#删除某个数据库或表，或某个表的一列的数据
rename database <dbname1> to <dbname2>;
rename table <tablename1> to <tablename2>;
#更改数据库名或者表名
union select
#联合查找，要求两个表的列数相同
order by <列名/列数>
#根据某一列排序
prepare <自定义语句名> from '<mysql语句>'
#准备一条SQL语句，并分配给这条SQL语句一个名字供之后调用
execute <上面定义的语句名>
#执行命令
deallocate prepare <上面定义的语句名>
#释放命令
set @<变量名>=<键值>
#创建变量，如：set @a=1;set @b='select ?';mysql中可以用问号?来代替变量所在的位置如调用前面定义的语句即可prepare test form @b using @a;
using @<变量名>
