# mysql 中 group_concat()用法

原文网址【博客园】：https://www.cnblogs.com/nww57/p/5283050.html

# 用处：将所选列的值连接在一起

# 基本语法：

group_concat([DISTINCT] <要连接的字段> [Order BY \**排序字段\** ASC/DESC] [Separator '分隔符'])

#<要连接的字段>即为表的列名，是必填项，其他都是选填，默认的分隔符是逗号,

# 举例：

代码为：

1' union select 1,group_concat(table_name) collate utf8_general_ci from information_schema.tables where table_schema =database()#

#即将information_schema.tables这张表中table_schema =database()的table_name全部列连接在一起显示出来

# 初始数据：

​

以name分组，将money打印在一行，默认','分隔：

select aa.namename , GROUP_CONCAT( money ) money from aa GROUP BY aa.name

*以name分组，将money打印在一行，用'；'分隔*

select aa.namename , GROUP_CONCAT( money separator ';') money from aa GROUP BY aa.name

****以name分组，去掉冗余的money，用'；'分隔******

select aa.namename , GROUP_CONCAT(DISTINCT money separator ';') money from aa GROUP BY aa.name
